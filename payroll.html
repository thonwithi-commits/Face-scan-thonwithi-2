/** * ฟังก์ชันหลักสำหรับรับข้อมูลจากแอป iPad (index.html) และหน้าจัดการ (payroll.html) */ function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- 1. ส่วนบันทึกข้อมูล (แอปเครื่องแสกน) --- // บันทึกเวลาเข้า-ออกงาน if (data.action == 'log') { var sheet = ss.getSheetByName('Logs'); // หากไม่มีชีต Logs ให้สร้างขึ้นมาอัตโนมัติ if (!sheet) { sheet = ss.insertSheet('Logs'); sheet.appendRow(['Timestamp', 'Name', 'Type', 'Lat', 'Lng']); } sheet.appendRow([new Date(), data.name, data.type, data.lat, data.lng]); return ContentService.createTextOutput("Success"); } // ลงทะเบียนใบหน้าพนักงานใหม่ if (data.action == 'register') { var sheet = ss.getSheetByName('Users'); if (!sheet) { sheet = ss.insertSheet('Users'); sheet.appendRow(['Name', 'Descriptor']); } sheet.appendRow([data.name, JSON.stringify(data.descriptor)]); return ContentService.createTextOutput("Success"); } // --- 2. ส่วนดึงข้อมูล (หน้า Payroll Dashboard) --- // ดึงประวัติการลงเวลาทั้งหมด if (data.action == 'getLogs') { var sheet = ss.getSheetByName('Logs'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var logs = rows.slice(1).map(function(r) { return { timestamp: Utilities.formatDate(new Date(r[0]), "GMT+7", "yyyy-MM-dd HH:mm:ss"), name: r[1], type: r[2] }; }); return returnJson(logs); } // ดึงรายชื่อพนักงานและตั้งค่าอัตราจ้าง/เวลาเข้างาน (จากชีต Settings) if (data.action == 'getEmployeeSettings') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var settings = rows.slice(1).map(function(r) { return { name: r[0], dept: r[1], empType: r[2], // "รายวัน" หรือ "รายชั่วโมง" rate: r[3], startTime: r[4], // เวลาเข้างานปกติ เช่น "08:00" advance: r[5] || 0 // ดึงยอดเบิกล่วงหน้า (คอลัมน์ F) }; }); return returnJson(settings); } // อัปเดตยอดเบิกล่วงหน้าจากหน้าเว็บ if (data.action == 'updateAdvance') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson({status: "error", message: "ไม่พบแผ่นงาน Settings"}); var rows = sheet.getDataRange().getValues(); // วนหาชื่อพนักงานที่ตรงกัน for (var i = 1; i < rows.length; i++) { if (rows[i][0].toString().trim() === data.name.trim()) { // อัปเดตยอดเงินในคอลัมน์ F (คอลัมน์ที่ 6) sheet.getRange(i + 1, 6).setValue(data.advance); break; } } return returnJson({status: "success"}); } // ดึงข้อมูลใบหน้าพนักงานสำหรับแอปแสกน if (data.action == 'getKnownFaces') { var sheet = ss.getSheetByName('Users'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var faces = rows.slice(1).map(function(r) { return { label: r[0], descriptor: JSON.parse(r[1]) }; }); return returnJson(faces); } } /** * ฟังก์ชันช่วยเหลือสำหรับการตอบกลับเป็น JSON */ function returnJson(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } /** * ฟังก์ชันสำหรับตรวจสอบความพร้อมของ API ผ่านบราวเซอร์ */ function doGet(e) { return HtmlService.createHtmlOutput("

API ระบบลงเวลา Thonwithi พร้อมใช้งาน

สถานะ: ปกติ

"); }
