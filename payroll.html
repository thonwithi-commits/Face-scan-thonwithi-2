/** * ฟังก์ชันหลักสำหรับรับข้อมูลจากแอป iPad (index.html) และหน้าจัดการ (payroll.html) */ function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.getActiveSpreadsheet(); // --- 1. ส่วนบันทึกข้อมูล (แอปเครื่องแสกน) --- if (data.action == 'log') { var sheet = ss.getSheetByName('Logs'); if (!sheet) { sheet = ss.insertSheet('Logs'); sheet.appendRow(['Timestamp', 'Name', 'Type', 'Lat', 'Lng']); } sheet.appendRow([new Date(), data.name, data.type, data.lat, data.lng]); return ContentService.createTextOutput("Success"); } if (data.action == 'register') { var sheet = ss.getSheetByName('Users'); if (!sheet) { sheet = ss.insertSheet('Users'); sheet.appendRow(['Name', 'Descriptor']); } sheet.appendRow([data.name, JSON.stringify(data.descriptor)]); return ContentService.createTextOutput("Success"); } // --- 2. ส่วนดึงข้อมูล (หน้า Payroll Dashboard) --- if (data.action == 'getLogs') { var sheet = ss.getSheetByName('Logs'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var logs = rows.slice(1).map(function(r) { return { timestamp: Utilities.formatDate(new Date(r[0]), "GMT+7", "yyyy-MM-dd HH:mm:ss"), name: r[1], type: r[2] }; }); return returnJson(logs); } if (data.action == 'getEmployeeSettings') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var settings = rows.slice(1).map(function(r) { return { name: r[0], dept: r[1], empType: r[2], rate: r[3], startTime: r[4], advance: r[5] || 0, // ดึงยอดเบิกล่วงหน้า (คอลัมน์ F) bonus: r[6] || 0 // ดึงยอดโบนัส (คอลัมน์ G) }; }); return returnJson(settings); } if (data.action == 'updateAdvance') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson({status: "error", message: "ไม่พบแผ่นงาน Settings"}); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0].toString().trim() === data.name.trim()) { sheet.getRange(i + 1, 6).setValue(data.advance); break; } } return returnJson({status: "success"}); } // อัปเดตยอดโบนัสให้ "ทุกคน" เท่าๆ กัน if (data.action == 'updateBonusAll') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson({status: "error", message: "ไม่พบแผ่นงาน Settings"}); var lastRow = sheet.getLastRow(); // อัปเดตคอลัมน์ G (คอลัมน์ที่ 7) ของพนักงานทุกคน if (lastRow > 1) { sheet.getRange(2, 7, lastRow - 1, 1).setValue(data.bonusPerPerson); } return returnJson({status: "success"}); } // อัปเดตยอดโบนัสรายบุคคล if (data.action == 'updateBonusSingle') { var sheet = ss.getSheetByName('Settings'); if (!sheet) return returnJson({status: "error", message: "ไม่พบแผ่นงาน Settings"}); var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0].toString().trim() === data.name.trim()) { sheet.getRange(i + 1, 7).setValue(data.bonus); break; } } return returnJson({status: "success"}); } if (data.action == 'getKnownFaces') { var sheet = ss.getSheetByName('Users'); if (!sheet) return returnJson([]); var rows = sheet.getDataRange().getValues(); var faces = rows.slice(1).map(function(r) { return { label: r[0], descriptor: JSON.parse(r[1]) }; }); return returnJson(faces); } } function returnJson(data) { return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return HtmlService.createHtmlOutput("
